```bash
module top_module( 
    input x3,
    input x2,
    input x1,  // three inputs
    output f   // one output
);
    
    assign f = (~x3 & x2 & ~x1) | (~x3 & x2 & x1) | (x3 & ~x2 & x1) | (x3 & x2 & x1);

endmodule

module top_module ( input [1:0] A, input [1:0] B, output z ); 
    
    assign z = (~A[1] & ~A[0] & ~B[1] & ~B[0]) | (A[1] & ~A[0] & B[1] & ~B[0]) 
        | (~A[1] & A[0] & ~B[1] & B[0]) | (A[1] & A[0] & B[1] & B[0]);

endmodule

module top_module (input x, input y, output z);

    wire z1, z2, z3, z4;
    
    qa qa1 (x, y, z1);
    qb qb1 (x, y, z2);
    qa qa2 (x, y, z3);
    qb qb2 (x, y, z3);
    
    assign z = (z1 | z2) ^ (z3  & z4);
    
endmodule

module qa (input x, input y, output z);
    assign z = (x^y) & x;
    
endmodule

module qb (input x, input y, output z);
    assign z = (~x & ~y) | (x & y);
    
endmodule

module top_module (
    input too_cold,
    input too_hot,
    input mode,
    input fan_on,
    output heater,
    output aircon,
    output fan
); 
    
    assign heater = too_cold & mode;
    assign aircon = too_hot & ~mode;
    assign fan = heater | aircon | fan_on;

endmodule



```
