```bash
module top_module(
    input [31:0] a,
    input [31:0] b,
    input sub,
    output [31:0] sum
);
    
    wire carry; wire [31:0] xorb;
    
    add16 top (.a(a[15:0]), .b(xorb[15:0]), .cin(sub), .cout(carry), .sum(sum[15:0]));
    add16 bot (.a(a[31:16]), .b(xorb[31:16]), .cin(carry), .sum(sum[31:16]));

    assign xorb = {32{sub}}^b;

endmodule

// synthesis verilog_input_version verilog_2001
module top_module(
    input a,
    input b,
    input sel_b1,
    input sel_b2,
    output wire out_assign,
    output reg out_always   ); 
    
    always @(*) begin
        if ((sel_b1 & sel_b2) == 1'b1) out_always = b;
        else out_always = a;
    end
    
    assign out_assign = ((sel_b1 & sel_b2) == 1'b1)? b : a;

endmodule


```
