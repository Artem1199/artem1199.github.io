```bash
module top_module (
    input [7:0] a,
    input [7:0] b,
    output [7:0] s,
    output overflow
); //
    wire [7:0]cout;
    adder adder1(a[0], b[0], 1'b0, cout[0], s[0]);
    adder adder2(a[1], b[1], cout[0], cout[1], s[1]);
    adder adder3(a[2], b[2], cout[1], cout[2], s[2]);
    adder adder4(a[3], b[3], cout[2], cout[3], s[3]);
    adder adder5(a[4], b[4], cout[3], cout[4], s[4]);
    adder adder6(a[5], b[5], cout[4], cout[5], s[5]);
    adder adder7(a[6], b[6], cout[5], cout[6], s[6]);
    adder adder8(a[7], b[7], cout[6], cout[7], s[7]);
    
    assign overflow = (~a[7] & ~b[7] & s[7]) | (a[7] & b[7] & ~s[7]); 
    
    endmodule
        
module adder( 
    input a, b, cin,
    output cout, sum );
    
    assign sum = ~cin & (a^b) | cin & ~(a^b);
    assign cout = (~cin & a & b) | (cin & ~a & b) | (cin & a & ~b) | (cin & a & b);

endmodule


```
