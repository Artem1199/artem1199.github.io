Completed the following module, during the process I accidently set 'assign' in the always block. Which generated the 'Procedural Continuous Assignment to register is not supported File' failure.

```bash
module top_module(
    input [31:0] a,
    input [31:0] b,
    output [31:0] sum
);
    wire topsel;
    wire [15:0] midtomulti, bottomulti;
    
    add16 top_adder (.a(a[15:0]), .b(b[15:0]), .cin(0), .cout(topsel), .sum(sum[15:0]));
    add16 mid_adder (.a(a[31:16]), .b(b[31:16]), .cin(0), .sum(midtomulti));
    add16 bot_adder (.a(a[31:16]), .b(b[31:16]), .cin(1), .sum(bottomulti));
    
    always @(*) begin
        case (topsel) 
            1'b0: sum[31:16] = midtomulti;
            1'b1: sum[31:16] = bottomulti;
           default:
               sum[31:16] = 1'bx;
       endcase
    end
    
endmodule
```
